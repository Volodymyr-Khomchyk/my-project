git init - ініціалізація репозиторія

 

git add . - додати всі зміни до індексу

 

git commit - створити коміт на основі проіндексованих змін

            -m "Назва коміту" - скорочений варіант

 

git log - історія комітів

        --oneline - в одну строку

 

git diff - переглянути незафіксовані змін

 

git diff --staged - переглянути зафіксовані зміни

 

///////////////////////////

 

git remote - команда для керування віддаленими репозиторіями

git remote -v - перевірка підв'язки віддаленого репозиторія

 

git push origin <гілка> - завантажити зміни на віддалений репозиторій

 

git clone <адреса> - зклонувати віддалений репозиторій до себе на пк

 

//////////////////////////

 

git restore - відновлює стан файлу до останнього коміту

           --staged - повертає файл з індексації (зміни не будуть індексовані)


////////////////////////

 

git branch <ім'я гілки> - команда для створення нової гілки

           -m - перейменовує гілку


git branch (без аргументів) - показує всі наявні гілки в поточному репозиторії
 
git checkout <ім'я гілки>

git switch <ім'я гілки>    - команди, які перемикають гілку

//////////////////////////////////

git push origin dev    // завантажує гілку dev на гітхаб

git push origin test   // завантажує гілку test на гітхаб

git push origin main // завантажує гілку main на гітхаб
